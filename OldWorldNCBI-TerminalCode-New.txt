#transferring data from NCBI
curl -o datasets 'https://ftp.ncbi.nlm.nih.gov/pub/datasets/command-line/v2/mac/datasets'
curl -o dataformat 'https://ftp.ncbi.nlm.nih.gov/pub/datasets/command-line/v2/mac/dataformat'
chmod +x datasets dataformat

#Downloading MHC genes
./datasets download gene symbol MAMU-A MAMU-B MAMU-DRB1 MAMU-DQA1 MAMU-DQB1 MAMU-DQA1 MAMU-DPB MAMU-DRB5 --taxon "Macaca mulatta" 
./datasets download gene symbol HLA-A HLA-B HLA-C HLA-DRB1 HLA-DQB1 HLA-DQA1 --taxon "Homo sapiens" 
./datasets download gene symbol LOC140708851 LOC103221783 MHC-DQB1 MHC-DOA MHC-DRA MHC-DMA --taxon "Chlorocebus sabaeus"
